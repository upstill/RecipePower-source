def self.load str
  hsh = {}
  str.split("\t").each { |nv| md = nv.match( /([^:]*):(.*)/); hsh[md[1].to_sym] = md[2] }
  hsh[:klass] = hsh[:klass].constantize if hsh[:klass]
  hsh[:id] = hsh[:id].to_i if hsh[:id]
  self.new hsh
end

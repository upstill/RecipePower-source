def assert_words new
  wordlist = split + (new.is_a?(String) ? new.split : new.map(&:to_s))
  wordlist.compact.uniq*' '
end

def extensions_to_classes
  this = nil
  self.split('.').collect { |substr|
    this = this ? "#{this}-#{substr.downcase}" : substr.downcase
  }.join ' '
end

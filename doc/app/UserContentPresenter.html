def sibling_types
  case result_type
    when 'lists.owned'
      [ 'lists.collected' ]
    when 'lists.collected'
      [ 'lists.owned' ]
    when 'lists', 'feeds', 'friends', 'cookmarks' #, 'recipes'
      %w{ lists cookmarks feeds friends } - [result_type]
    else
      super
  end
end

def subtypes
  case result_type
    when 'lists'
      %w{ lists.owned lists.collected }
    when 'collection'
      %w{ cookmarks lists feeds friends }
    else
      super
  end
end

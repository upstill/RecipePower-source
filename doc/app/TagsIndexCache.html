def itemscope
  return @itemscope if @itemscope
  @itemscope = @tagtype ? Tag.where(tagtype: @tagtype) : Tag.unscoped
  if @batch
    first = (@batch.to_i-1) * 100
    @itemscope = @itemscope.order(id: :ASC).where("id >= #{first} and id < #{first+100}")
  end
  @itemscope
end

def sort sortfield = :id, descending = false
  @sortfield = sortfield.to_sym
  reverse = descending ? -1 : 1
  @data = @data.compact.sort { |us1, us2|
    comp =
    if us1[@sortfield] && us2[@sortfield]
      us1[@sortfield] <=> us2[@sortfield]
    elsif us1[@sortfield]
      1
    elsif us2[@sortfield]
      -1
    else
      0
    end
    comp * reverse
  }
end

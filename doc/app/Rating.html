def value_as_text
    if(self.scale_id.nil? || self.scale_val.nil?) 
       "<nil value>"
    else
       Scale.exists?(self.scale_id) ? 
       Scale.find(self.scale_id).value_as_text(self.scale_val) :
       "lost scale #{self.scale_id}"
    end
end

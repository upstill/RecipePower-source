def initialize val
  case val
    when nil
      super ''
    when Hash
      super val[:result_type] || ''
    when Symbol
      super val.to_s
    when Draper::Decorator
      super val.object.class.to_s
    when String
      super
    else
      super val.class.to_s
  end
end

def entity_params
  @entity_params ||=
      case root
        when '', nil
          { }
        when 'cookmarks'
          { entity_type: %w{ Recipe Site FeedEntry PageRef } }
        else
          { entity_type: self.model_name }
      end
end

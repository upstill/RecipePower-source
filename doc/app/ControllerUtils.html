def action_summary controller, action
  case controller
    when "users"
      case action
        when "collection"
          return "view someone's collection"
      end
    when "invitations"
      case action
        when "new"
          return "invite someone else"
      end
  end
  "#{action} #{controller}"
end

def with_format(format, &block)
  old_formats = formats
  self.formats = [format]
  result = block.call
  self.formats = old_formats
  result
end
